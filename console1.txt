SELECT s.name,
       s.identifier,
       s.user_id,
       ep.min_score,
       ep.max_score,
       ep.mean_score,
       ep.evaluation_count,
       ef.form_score,
       ev.email,
       ep.name,
       dt.eva_date_yyyymmdd,
       dt.eva_week_of_month,
       pt.name,
       ef.primary_resource_description,
       ef.resource_count,
       ef.id,
       ev.name,
       r.resource_create_date,
       r.quantity,
       dt.eva_month,
       ef.status,
       dt.eva_year,
       dt.eva_month_name_tr,
       ef.evaluator_comment,
       ev.id,
       ef.id,
       r.id,
       ef.primary_resource_embedded_id,
       r.embedded_id,
       mt2.text,
       mt1.text,
       s.group_name,
       ef.subject_group_name,
       r.description,
       ef.first_release_date,
       r.resource_uri,
       mt2.text,
       mt1.text,
       ef.first_release_date,
       ef.evaluation_date,
       ep1.resource_create_date,
       r.resource_group_id,
       r.group_name

FROM evaluator ev,
     eva_date_dimension dt,
     evaluation_process ep

         LEFT OUTER JOIN subject s ON (s.id = ep.subject_id)
         LEFT OUTER JOIN process_template pt ON (pt.id = ep.process_template_id),
     evaluation_form ef
         RIGHT OUTER JOIN resource r ON (ef.id = r.evaluation_form_id)
         LEFT OUTER JOIN meta_data_type mt1 ON (mt1.meta_key = ef.evaluation_result_meta_data)
         LEFT OUTER JOIN meta_data_type mt2 ON (mt2.meta_key = ef.evaluation_reason_meta_data)
         RIGHT OUTER JOIN dbo.resource ep1 ON (ef.id = ep1.evaluation_form_id)


WHERE (ep.id = ef.evaluation_process_id
    AND ev.id = ef.evaluator_id
    AND ef.primary_resource_embedded_id = r.embedded_id
    AND dt.id = ef.evaluation_date_dim_id
    AND ef.primary_resource_embedded_id = ep1.embedded_id)
    AND ((pt.name = 'Inbound Degerlendirme SÃ¼reci'
    AND dt.eva_year = 2022
    AND dt.eva_month_name_tr = 'Ocak'
    AND ef.status = 'RELEASED'
    AND s.name
    IN ('142 142', '1001 1001')))
